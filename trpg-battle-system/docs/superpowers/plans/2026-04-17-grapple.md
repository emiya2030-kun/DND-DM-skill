# Grapple Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a complete combat grapple loop: start grapple, enforce grappled movement restrictions, drag the target while moving at half speed, let the target escape with an action, and auto-release invalid grapples.

**Architecture:** Represent the grappled target with the existing sourced condition (`grappled:<source_id>`) and store the active maintained grapple on the grappler in `combat_flags["active_grapple"]`. Build explicit grapple services and runtime commands, then integrate dragging and release logic into the existing movement, condition, start-turn, and encounter-state projection pipelines.

**Tech Stack:** Python 3, `unittest`, existing encounter repositories, runtime command dispatcher, movement services, condition services.

```python
def resolve_grapple_save_dc(actor: EncounterEntity) -> dict[str, Any]:
    monk = ((actor.class_features or {}).get("monk") or {}) if isinstance(actor.class_features, dict) else {}
    martial_arts = monk.get("martial_arts") if isinstance(monk, dict) else None
    ability_used = "dex" if isinstance(martial_arts, dict) and martial_arts.get("grapple_dc_ability") == "dex" else "str"
    ability_mod = int(actor.ability_mods.get(ability_used, 0))
    proficiency_bonus = int(actor.proficiency_bonus or 0)
    return {
        "dc": 8 + ability_mod + proficiency_bonus,
        "ability_used": ability_used,
        "breakdown": {
            "base": 8,
            "ability_mod": ability_mod,
            "proficiency_bonus": proficiency_bonus,
        },
    }


def build_active_grapple_payload(*, actor: EncounterEntity, target: EncounterEntity, save_dc: dict[str, Any]) -> dict[str, Any]:
    return {
        "target_entity_id": target.entity_id,
        "escape_dc": save_dc["dc"],
        "dc_ability_used": save_dc["ability_used"],
        "movement_speed_halved": True,
        "source_condition": f"grappled:{actor.entity_id}",
    }


def extract_grapple_source_id(entity: EncounterEntity) -> str | None:
    for condition in entity.conditions:
        if isinstance(condition, str) and condition.startswith("grappled:"):
            return condition.split(":", 1)[1].strip() or None
    return None
```

```python
class UseGrapple:
    def execute(self, *, encounter_id: str, actor_id: str, target_id: str) -> dict[str, Any]:
        encounter = self._get_encounter_or_raise(encounter_id)
        actor = self._get_actor_or_raise(encounter, actor_id)
        target = self._get_target_or_raise(encounter, target_id)
        self._ensure_actor_turn(encounter, actor_id)
        self._ensure_action_available(actor)
        self._ensure_target_is_enemy(actor, target)
        self._ensure_target_within_reach(actor, target)
        self._ensure_actor_has_no_active_grapple(actor)

        save_dc = resolve_grapple_save_dc(actor)
        target_save_total = max(
            int(target.ability_mods.get("str", 0)) + int(target.proficiency_bonus if "str" in target.save_proficiencies else 0),
            int(target.ability_mods.get("dex", 0)) + int(target.proficiency_bonus if "dex" in target.save_proficiencies else 0),
        )
        actor.action_economy["action_used"] = True
        if target_save_total < save_dc["dc"]:
            target.conditions.append(f"grappled:{actor.entity_id}")
            actor.combat_flags["active_grapple"] = build_active_grapple_payload(actor=actor, target=target, save_dc=save_dc)
            status = "grappled"
        else:
            status = "saved"
        self.encounter_repository.save(encounter)
        return {"result": {"status": status}, "encounter_state": self.get_encounter_state.execute(encounter_id)}
```

```python
class EscapeGrapple:
    def execute(self, *, encounter_id: str, actor_id: str) -> dict[str, Any]:
        encounter = self._get_encounter_or_raise(encounter_id)
        actor = self._get_actor_or_raise(encounter, actor_id)
        self._ensure_actor_turn(encounter, actor_id)
        self._ensure_action_available(actor)
        source_id = extract_grapple_source_id(actor)
        if source_id is None:
            raise ValueError("actor_not_grappled")
        grappler = self._get_actor_or_raise(encounter, source_id)
        active_grapple = grappler.combat_flags.get("active_grapple", {})
        escape_dc = int(active_grapple.get("escape_dc", 0))
        athletics = int(actor.skill_modifiers.get("athletics", actor.ability_mods.get("str", 0)))
        acrobatics = int(actor.skill_modifiers.get("acrobatics", actor.ability_mods.get("dex", 0)))
        actor.action_economy["action_used"] = True
        if max(athletics, acrobatics) >= escape_dc:
            actor.conditions = [condition for condition in actor.conditions if condition != f"grappled:{source_id}"]
            grappler.combat_flags.pop("active_grapple", None)
            status = "escaped"
        else:
            status = "still_grappled"
        self.encounter_repository.save(encounter)
        return {"result": {"status": status}, "encounter_state": self.get_encounter_state.execute(encounter_id)}
```

```python
def release_grapple_if_invalid(encounter: Encounter, grappler_id: str) -> bool:
    grappler = encounter.entities.get(grappler_id)
    if grappler is None:
        return False
    active_grapple = grappler.combat_flags.get("active_grapple", {}) if isinstance(grappler.combat_flags, dict) else {}
    target_id = active_grapple.get("target_entity_id")
    if not isinstance(target_id, str):
        return False
    target = encounter.entities.get(target_id)
    should_release = (
        target is None
        or "grappled:%s" % grappler_id not in getattr(target, "conditions", [])
        or ConditionRuntime(grappler.conditions).has("incapacitated")
    )
    if not should_release:
        return False
    if target is not None:
        target.conditions = [condition for condition in target.conditions if condition != f"grappled:{grappler_id}"]
    grappler.combat_flags.pop("active_grapple", None)
    return True
```

Call it after apply/remove in `UpdateConditions.execute(...)`, using either the updated grappler or the grappler extracted from the removed condition.
